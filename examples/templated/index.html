The above div is a map initialized like so:

// "import" the namespace
var mm = com.modestmaps;

var container = document.getElementById('container');

map = new mm.Map('map', new mm.TemplatedMapProvider('http://tile.openstreetmap.org/{Z}/{X}/{Y}.png'), new mm.Point(container.offsetWidth, container.offsetHeight))

var f = new mm.Follower(map, new mm.Location(37.811530, -122.2666097), '° Broadway and Grand');

map.setCenterZoom(new mm.Location(37.811530, -122.2666097), 14);

window.onresize = function() {
    map.setSize(container.offsetWidth, container.offsetHeight);
};
